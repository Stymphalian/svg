Regenerate the docs gulp docs

// create a new <svg> element under the DOM element specified by the 'div_id'
// with a widht of 400 and height of 300
var canvas = LIBRARY.svg("div_id",400,300);

// create a <circle> element at x=5,y=5,r=40 and add it as a child 
// of the canvas node.
var circle = LIBRARY.circle(canvas,5,5,40);
LIBRARY.attr(circle, {"color":red});

// circle is a DOM node, and therefore has access to 
// all the methods provided by the DOM interface
circle.addEventListener("click",function(){},false);
